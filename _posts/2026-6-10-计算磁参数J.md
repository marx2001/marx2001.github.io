## <center>说明</center>

原创脚本。原则上可用于任意异质结、单层、块体体系的磁参数J的计算。测试过三角晶格体系。

## <center>方法</center>  

```python

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Generate Heisenberg energy-mapping coefficients for single-layer or bilayer magnetic systems.

Hamiltonian convention used here:
    E = E0 - sum_{<i,j>, shell/type} J_type_shell * S_i * S_j
where S_i = +1 or -1 for collinear magnetic configurations.

For every generated magnetic configuration, the coefficient of a J parameter is:
    c = - sum_{bonds in that pair type and neighbour shell} S_i * S_j

The script:
  1. Reads a VASP POSCAR/CONTCAR.
  2. Selects magnetic atoms by element or explicit POSCAR indices.
  3. Finds unique magnetic bonds under periodic boundary conditions.
  4. Splits bonds into neighbour shells separately for each pair type.
  5. Generates linearly independent collinear spin configurations.
  6. Writes the coefficient matrix and equations for fitting J.

Typical use for a Cr/Ru bilayer heterostructure:
    python3 generate_J_mapping_coefficients.py POSCAR \
        --mag-elements Cr Ru \
        --group-by element \
        --max-shells 3 \
        --pbc 1 1 0 \
        --n-configs 12 \
        --outdir J_fit_output

Typical use for a single magnetic layer with one magnetic element M:
    python3 generate_J_mapping_coefficients.py POSCAR \
        --mag-elements Cr \
        --group-by element \
        --max-shells 3 \
        --pbc 1 1 0 \
        --n-configs 8 \
        --outdir J_fit_output

Author: generated for atomistic spin-model J fitting workflows. （Ruixiao Ma）
"""

import argparse

import numpy as np


@dataclass
class Atom:
    index0: int          # 0-based POSCAR atom index
    element: str
    frac: np.ndarray     # fractional coordinates, wrapped to [0,1)
    cart: np.ndarray     # Cartesian coordinates in Angstrom
    layer: str = "L1"


@dataclass
class Bond:
    i: int               # index in mag_atoms list
    j: int               # index in mag_atoms list
    shift: Tuple[int, int, int]
    distance: float
    pair_label: str
    shell: int = 0


def parse_poscar(path: str):
    with open(path, "r", encoding="utf-8", errors="ignore") as f:
        lines = [line.rstrip() for line in f if line.strip()]
    if len(lines) < 8:
        raise ValueError("POSCAR seems too short.")

    scale = float(lines[1].split()[0])
    lattice = np.array([[float(x) for x in lines[i].split()[:3]] for i in range(2, 5)], dtype=float)
    lattice *= scale

    # VASP5: line 5 elements, line 6 counts. VASP4 has no element names.
    line5 = lines[5].split()
    try:
        counts = [int(x) for x in line5]
        elements = [f"X{i+1}" for i in range(len(counts))]
        coord_line_idx = 6
    except ValueError:
        elements = line5
        counts = [int(x) for x in lines[6].split()]
        coord_line_idx = 7

    if lines[coord_line_idx].lower().startswith("s"):
        coord_line_idx += 1

    coord_type = lines[coord_line_idx].strip().lower()[0]
    coord_start = coord_line_idx + 1
    natoms = sum(counts)
    coord_lines = lines[coord_start:coord_start + natoms]
    if len(coord_lines) < natoms:
        raise ValueError("Not enough coordinate lines in POSCAR.")

    inv_lattice = np.linalg.inv(lattice)
    atoms: List[Atom] = []
    idx = 0
    for el, num in zip(elements, counts):
        for _ in range(num):
            vals = [float(x) for x in coord_lines[idx].split()[:3]]
            arr = np.array(vals, dtype=float)
            if coord_type in ["d", "f"]:  # Direct/Fractional
                frac = arr.copy()
                cart = frac @ lattice
            elif coord_type in ["c", "k"]:  # Cartesian/Kpoints style label in VASP
                cart = arr * scale if abs(scale - 1.0) > 1e-12 else arr
                frac = cart @ inv_lattice
            else:
                raise ValueError(f"Unknown coordinate type line: {lines[coord_line_idx]}")
            frac_wrapped = frac - np.floor(frac)
            cart_wrapped = frac_wrapped @ lattice
            atoms.append(Atom(index0=idx, element=el, frac=frac_wrapped, cart=cart_wrapped))
            idx += 1

    return lattice, elements, counts, atoms


def parse_index_list(s: str) -> List[int]:
    """Parse comma-separated 1-based or 0-based indices. Ranges allowed, e.g. 1-4,7,8."""
    out = []
    for part in s.replace(" ", "").split(","):
        if not part:
            continue
        if "-" in part:
            a, b = part.split("-", 1)
            a, b = int(a), int(b)
            out.extend(range(a, b + 1))
        else:
            out.append(int(part))
    return out


def assign_layers_by_z(mag_atoms: List[Atom], n_layers: int):
    if n_layers <= 1:
        for a in mag_atoms:
            a.layer = "L1"
        return
    zvals = np.array([a.frac[2] for a in mag_atoms])
    order = np.argsort(zvals)
    sorted_z = zvals[order]

    # Split sorted atoms into n_layers clusters using the largest internal gaps.
    gaps = sorted_z[1:] - sorted_z[:-1]
    if len(gaps) < n_layers - 1:
        raise ValueError("Not enough magnetic atoms to assign layers.")
    cut_positions = np.argsort(gaps)[-(n_layers - 1):] + 1
    cut_positions = sorted(cut_positions.tolist())
    groups = np.split(order, cut_positions)
    # Label bottom to top as L1, L2, ... according to average z.
    groups_sorted = sorted(groups, key=lambda g: float(np.mean(zvals[g])))
    for lid, group in enumerate(groups_sorted, start=1):
        for idx in group:
            mag_atoms[int(idx)].layer = f"L{lid}"


def get_group_label(atom: Atom, group_by: str) -> str:
    if group_by == "element":
        return atom.element
    if group_by == "layer":
        return atom.layer
    if group_by == "element_layer":
        return f"{atom.element}_{atom.layer}"
    raise ValueError(f"Unknown group_by: {group_by}")


def pair_label(a: Atom, b: Atom, group_by: str) -> str:
    ga = get_group_label(a, group_by)
    gb = get_group_label(b, group_by)
    return "-".join(sorted([ga, gb]))


def canonical_bond_key(i: int, j: int, shift: Tuple[int, int, int]) -> Tuple:
    tx, ty, tz = shift
    rep1 = ((i, 0, 0, 0), (j, tx, ty, tz))
    rep2 = ((j, 0, 0, 0), (i, -tx, -ty, -tz))
    return min(rep1, rep2)


def find_unique_bonds(
    lattice: np.ndarray,
    mag_atoms: List[Atom],
    pbc: Tuple[int, int, int],
    max_image: int,
    group_by: str,
) -> List[Bond]:
    shifts = []
    ranges = []
    for flag in pbc:
        ranges.append(range(-max_image, max_image + 1) if flag else range(0, 1))
    for tx, ty, tz in itertools.product(*ranges):
        shifts.append((tx, ty, tz))

    bonds = []
    seen = set()
    frac_coords = np.array([a.frac for a in mag_atoms])

    for i, ai in enumerate(mag_atoms):
        for j, aj in enumerate(mag_atoms):
            for sh in shifts:
                if i == j and sh == (0, 0, 0):
                    continue
                key = canonical_bond_key(i, j, sh)
                if key in seen:
                    continue
                seen.add(key)
                dfrac = frac_coords[j] + np.array(sh, dtype=float) - frac_coords[i]
                dcart = dfrac @ lattice
                dist = float(np.linalg.norm(dcart))
                if dist < 1e-8:
                    continue
                bonds.append(Bond(i=i, j=j, shift=sh, distance=dist, pair_label=pair_label(ai, aj, group_by)))
    return bonds


def assign_shells(
    bonds: List[Bond],
    max_shells: int,
    shell_tol: float,
    max_distance: Optional[float] = None,
) -> Tuple[List[Bond], Dict[str, List[float]]]:
    kept: List[Bond] = []
    shell_centers: Dict[str, List[float]] = {}
    by_pair: Dict[str, List[Bond]] = {}
    for b in bonds:
        if max_distance is not None and b.distance > max_distance:
            continue
        by_pair.setdefault(b.pair_label, []).append(b)

    for plabel, blist in by_pair.items():
        blist_sorted = sorted(blist, key=lambda x: x.distance)
        centers: List[float] = []
        shell_lists: List[List[Bond]] = []
        for b in blist_sorted:
            assigned = False
            for sidx, cen in enumerate(centers):
                if abs(b.distance - cen) <= shell_tol:
                    shell_lists[sidx].append(b)
                    # update center as mean for stability
                    centers[sidx] = float(np.mean([x.distance for x in shell_lists[sidx]]))
                    assigned = True
                    break
            if not assigned:
                centers.append(b.distance)
                shell_lists.append([b])
            # Stop only after collecting all bonds belonging to max_shells shells.
        order = np.argsort(centers).tolist()
        centers_sorted = [centers[k] for k in order]
        shell_lists_sorted = [shell_lists[k] for k in order]
        shell_centers[plabel] = centers_sorted[:max_shells]
        for shell_no, slist in enumerate(shell_lists_sorted[:max_shells], start=1):
            for b in slist:
                b.shell = shell_no
                kept.append(b)
    kept = sorted(kept, key=lambda b: (b.pair_label, b.shell, b.distance, b.i, b.j, b.shift))
    return kept, shell_centers


def parameter_names(bonds: List[Bond]) -> List[str]:
    params = sorted(set(f"J{b.shell}_{b.pair_label.replace('-', '')}" for b in bonds))
    # Natural sort by pair label then shell if possible
    def keyfn(name):
        # J1_CrRu -> (CrRu, 1)
        left, pair = name.split("_", 1)
        shell = int(left[1:])
        return (pair, shell)
    return sorted(params, key=keyfn)


def coefficient_row(spins: np.ndarray, bonds: List[Bond], params: List[str]) -> List[int]:
    acc = {p: 0 for p in params}
    for b in bonds:
        pname = f"J{b.shell}_{b.pair_label.replace('-', '')}"
        acc[pname] += int(spins[b.i] * spins[b.j])
    # Hamiltonian E = E0 - sum J SiSj, so coefficient is -sum SiSj.
    return [-acc[p] for p in params]


def config_name(k: int) -> str:
    return f"E{k}"


def generate_candidate_spins(mag_atoms: List[Atom], group_by_for_afm: str, seed: int, n_random: int) -> List[Tuple[str, np.ndarray]]:
    """Generate candidate collinear spin configurations.

    Priority in this version:
      1. FM is always E1 and has the highest priority.
      2. The next candidates are true AFM / compensated AFM-like configurations whenever possible:
         - inter-group/layer AFM, e.g. Cr up + Ru down;
         - exact 50/50 AFM patterns inside each group/layer when the group has an even number of atoms;
         - exact 50/50 AFM patterns in the whole magnetic supercell when the total number is even.
      3. Only if these AFM-priority candidates cannot provide full column rank, random non-AFM
         fallback configurations are used.

    Note:
      For an even number of magnetic atoms in a group, AFM-like means exactly N/2 up and N/2 down.
      The previous version used some "near-balanced" patterns, which could produce 4 up / 2 down
      states for a 6-atom layer. This version avoids that in the AFM-priority pool.
    """
    n = len(mag_atoms)
    rng = np.random.default_rng(seed)
    candidates: List[Tuple[str, np.ndarray]] = []

    def add_candidate(name: str, spins: np.ndarray, seen: set):
        spins = np.array(spins, dtype=int)
        # Keep FM as all-up. For other patterns, remove global spin-reversal duplicates.
        if name != "FM" and spins[0] == -1:
            spins = -spins
        key = tuple(spins.tolist())
        if key not in seen:
            seen.add(key)
            candidates.append((name, spins.copy()))

    def exact_balanced_patterns_for_indices(indices: List[int], max_patterns: int = 200) -> List[np.ndarray]:
        """Return exact 50/50 up/down patterns for selected indices if possible.

        The returned arrays have length len(indices). If len(indices) is odd, this returns
        the closest compensated patterns with floor(N/2) down spins, but these are used
        only when exact compensation is impossible.
        """
        m = len(indices)
        if m <= 1:
            return [np.ones(m, dtype=int)]

        n_down = m // 2
        patterns: List[np.ndarray] = []

        # Deterministic literature-like patterns first.
        deterministic_down_sets = []
        deterministic_down_sets.append(tuple(range(1, m, 2)))              # up/down alternating
        deterministic_down_sets.append(tuple(range(n_down, m)))            # first half up, second half down
        deterministic_down_sets.append(tuple(range(0, m, 2)))              # opposite alternating
        deterministic_down_sets.append(tuple([k for k in range(m) if (k // 2) % 2 == 1]))

        seen_local = set()
        for ds in deterministic_down_sets:
            ds = tuple(sorted(set(ds)))
            if len(ds) != n_down:
                continue
            if ds in seen_local:
                continue
            seen_local.add(ds)
            vals = np.ones(m, dtype=int)
            vals[list(ds)] = -1
            patterns.append(vals)

        # Exhaustive exact 50/50 combinations, capped to avoid huge candidate lists.
        for ds in itertools.combinations(range(m), n_down):
            if len(patterns) >= max_patterns:
                break
            if ds in seen_local:
                continue
            seen_local.add(ds)
            vals = np.ones(m, dtype=int)
            vals[list(ds)] = -1
            patterns.append(vals)

        return patterns

    seen = set()

    group_labels = [get_group_label(a, group_by_for_afm) for a in mag_atoms]
    unique_groups = sorted(set(group_labels))
    group_indices: Dict[str, List[int]] = {g: [i for i, lab in enumerate(group_labels) if lab == g] for g in unique_groups}

    # 1) Highest-priority reference: FM.
    add_candidate("FM", np.ones(n, dtype=int), seen)

    # 2) Inter-group/layer AFM references, e.g. Cr up/Ru down or top up/bottom down.
    # These are not necessarily zero total moment if the two groups have different sizes,
    # but they are the standard layer-AFM reference states.
    if len(unique_groups) >= 2:
        for g in unique_groups[1:]:
            spins = np.ones(n, dtype=int)
            for idx, lab in enumerate(group_labels):
                if lab == g:
                    spins[idx] = -1
            add_candidate(f"inter_AFM_{g}_down", spins, seen)

        spins = np.ones(n, dtype=int)
        base = unique_groups[0]
        for idx, lab in enumerate(group_labels):
            if lab != base:
                spins[idx] = -1
        add_candidate("inter_group_AFM", spins, seen)

    # 3) Exact AFM-like patterns within each group/layer, all other groups FM up.
    for g in unique_groups:
        patterns = exact_balanced_patterns_for_indices(group_indices[g], max_patterns=200)
        for pidx, vals in enumerate(patterns, start=1):
            spins = np.ones(n, dtype=int)
            for local_i, atom_i in enumerate(group_indices[g]):
                spins[atom_i] = vals[local_i]
            add_candidate(f"AFM_{g}_balanced{pidx}", spins, seen)

    # 4) Exact AFM-like patterns simultaneously applied to every group/layer.
    if len(unique_groups) >= 2:
        per_group_patterns = []
        for g in unique_groups:
            per_group_patterns.append(exact_balanced_patterns_for_indices(group_indices[g], max_patterns=20))
        for combo_idx, combo in enumerate(itertools.product(*per_group_patterns), start=1):
            if combo_idx > 400:
                break
            spins = np.ones(n, dtype=int)
            for g, vals in zip(unique_groups, combo):
                for local_i, atom_i in enumerate(group_indices[g]):
                    spins[atom_i] = vals[local_i]
            add_candidate(f"AFM_all_groups_balanced{combo_idx}", spins, seen)

            # Also add a version with every second group flipped. This can change interlayer
            # coefficients while preserving each group's exact zero net moment.
            spins2 = spins.copy()
            for gidx, g in enumerate(unique_groups):
                if gidx % 2 == 1:
                    for atom_i in group_indices[g]:
                        spins2[atom_i] *= -1
            add_candidate(f"AFM_all_groups_balanced{combo_idx}_interflip", spins2, seen)

    # 5) Whole-cell exact balanced AFM-like patterns.
    if n > 1:
        whole_patterns = exact_balanced_patterns_for_indices(list(range(n)), max_patterns=500)
        for pidx, vals in enumerate(whole_patterns, start=1):
            add_candidate(f"AFM_whole_balanced{pidx}", vals, seen)

    # 6) Balanced random AFM-like patterns.
    # For even n, these are exactly half up and half down. For odd n, closest possible.
    for _ in range(n_random):
        spins = np.ones(n, dtype=int)
        if n > 1:
            down = rng.choice(n, size=n // 2, replace=False)
            spins[down] = -1
        add_candidate("rand_balanced_AFM", spins, seen)

    # 7) Fully random non-AFM fallback patterns. These are intentionally last.
    for _ in range(n_random):
        spins = rng.choice([-1, 1], size=n).astype(int)
        add_candidate("rand_fallback_nonAFM", spins, seen)

    return candidates

def select_full_rank_configs(
    bonds: List[Bond],
    params: List[str],
    mag_atoms: List[Atom],
    n_configs: int,
    seed: int,
    group_by_for_afm: str,
) -> List[Tuple[str, np.ndarray, List[int]]]:
    unknowns = 1 + len(params)
    candidates = generate_candidate_spins(mag_atoms, group_by_for_afm, seed=seed, n_random=max(5000, 200 * n_configs))

    selected: List[Tuple[str, np.ndarray, List[int]]] = []

    def rank_of(sel):
        if not sel:
            return 0
        A = np.array([[1] + row[2] for row in sel], dtype=float)
        return int(np.linalg.matrix_rank(A, tol=1e-10))

    # Always include FM first if available. This is the highest-priority reference state.
    fm_added = False
    for name, spins in candidates:
        if name == "FM":
            selected.append((name, spins, coefficient_row(spins, bonds, params)))
            fm_added = True
            break
    if not fm_added:
        spins = np.ones(len(mag_atoms), dtype=int)
        selected.append(("FM", spins, coefficient_row(spins, bonds, params)))

    # Greedily add configurations in candidate priority order. Since the candidate list is
    # ordered as FM -> AFM-like -> balanced random AFM -> random fallback, the algorithm
    # uses AFM-like states whenever they increase rank, and only resorts to non-AFM states
    # if the AFM-like pool is insufficient.
    seen_spins = {tuple(selected[0][1].tolist())}
    for name, spins in candidates:
        if len(selected) >= n_configs and rank_of(selected) >= min(unknowns, n_configs):
            break
        key = tuple(spins.tolist())
        if key in seen_spins:
            continue
        coeff = coefficient_row(spins, bonds, params)
        r0 = rank_of(selected)
        test = selected + [(name, spins, coeff)]
        r1 = rank_of(test)
        if r1 > r0:
            selected.append((name, spins, coeff))
            seen_spins.add(key)
        if len(selected) >= n_configs and rank_of(selected) >= unknowns:
            break

    # If full rank has been reached but fewer than n_configs were requested, append extra
    # AFM-priority candidates first; random fallback candidates appear only after AFM-like
    # candidates in the candidate list.
    for name, spins in candidates:
        if len(selected) >= n_configs:
            break
        key = tuple(spins.tolist())
        if key in seen_spins:
            continue
        selected.append((name, spins, coefficient_row(spins, bonds, params)))
        seen_spins.add(key)

    if rank_of(selected) < unknowns:
        raise RuntimeError(
            f"Failed to generate full-column-rank configurations. Rank={rank_of(selected)}, unknowns={unknowns}. "
            f"Try increasing --n-configs, using a larger magnetic supercell, or changing --seed."
        )

    # Rename sequentially E1, E2, ... but keep original note in name suffix.
    renamed = []
    for k, (name, spins, coeff) in enumerate(selected[:n_configs], start=1):
        renamed.append((f"E{k}_{name}", spins, coeff))
    return renamed


def write_csv(path: str, header: List[str], rows: List[List]):
    with open(path, "w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(header)
        w.writerows(rows)


def equation_string(name: str, coeff: List[int], params: List[str]) -> str:
    terms = ["E0"]
    for c, p in zip(coeff, params):
        if c == 0:
            continue
        if c > 0:
            terms.append(f"+ {c}{p}")
        else:
            terms.append(f"- {abs(c)}{p}")
    return f"{name} = " + " ".join(terms)


def main():
    ap = argparse.ArgumentParser(description="Generate J-fitting magnetic configurations and coefficients.")
    ap.add_argument("poscar", help="Path to POSCAR/CONTCAR")
    ap.add_argument("--mag-elements", nargs="*", default=None, help="Magnetic elements, e.g. Cr Ru")
    ap.add_argument("--mag-indices", default=None, help="Explicit POSCAR atom indices. Default assumes 1-based, e.g. 1-6,13-18")
    ap.add_argument("--indices-zero-based", action="store_true", help="Interpret --mag-indices as 0-based")
    ap.add_argument("--group-by", choices=["element", "layer", "element_layer"], default="element", help="How to classify pair types")
    ap.add_argument("--n-layers", type=int, default=1, help="Number of magnetic layers for layer assignment")
    ap.add_argument("--pbc", nargs=3, type=int, default=[1, 1, 0], help="Periodic boundary flags, e.g. 1 1 0")
    ap.add_argument("--max-image", type=int, default=2, help="Maximum image shift searched in periodic directions")
    ap.add_argument("--max-shells", type=int, default=3, help="Number of neighbour shells per pair type")
    ap.add_argument("--shell-tol", type=float, default=0.08, help="Distance tolerance for grouping shells in Angstrom")
    ap.add_argument("--max-distance", type=float, default=None, help="Optional maximum bond distance in Angstrom")
    ap.add_argument("--n-configs", type=int, default=None, help="Number of magnetic configurations to output")
    ap.add_argument("--seed", type=int, default=20260610, help="Random seed for generated configurations")
    ap.add_argument("--outdir", default="J_fit_output", help="Output directory")
    args = ap.parse_args()

    lattice, elements, counts, atoms = parse_poscar(args.poscar)

    if args.mag_indices:
        raw = parse_index_list(args.mag_indices)
        if args.indices_zero_based:
            idx_set = set(raw)
        else:
            idx_set = set([x - 1 for x in raw])
        mag_atoms = [a for a in atoms if a.index0 in idx_set]
    elif args.mag_elements:
        mag_set = set(args.mag_elements)
        mag_atoms = [a for a in atoms if a.element in mag_set]
    else:
        raise ValueError("Please provide either --mag-elements or --mag-indices.")

    if not mag_atoms:
        raise ValueError("No magnetic atoms selected.")

    assign_layers_by_z(mag_atoms, args.n_layers)

    all_bonds = find_unique_bonds(lattice, mag_atoms, tuple(args.pbc), args.max_image, args.group_by)
    bonds, centers = assign_shells(all_bonds, args.max_shells, args.shell_tol, args.max_distance)
    params = parameter_names(bonds)
    unknowns = 1 + len(params)
    n_configs = args.n_configs if args.n_configs is not None else unknowns + 2

    # For group AFM candidates, use layer if n_layers>1, otherwise element.
    group_by_for_afm = "layer" if args.n_layers > 1 else "element"
    configs = select_full_rank_configs(bonds, params, mag_atoms, n_configs, args.seed, group_by_for_afm)

    os.makedirs(args.outdir, exist_ok=True)

    # Magnetic atom table
    atom_header = ["mag_index", "poscar_index_1based", "element", "layer", "frac_x", "frac_y", "frac_z"]
    atom_rows = []
    atom_labels = []
    for k, a in enumerate(mag_atoms):
        label = f"atom{a.index0+1}_{a.element}_{a.layer}"
        atom_labels.append(label)
        atom_rows.append([k, a.index0 + 1, a.element, a.layer, *[f"{x:.10f}" for x in a.frac]])
    write_csv(os.path.join(args.outdir, "magnetic_atoms.csv"), atom_header, atom_rows)

    # Bond table
    bond_header = ["pair_label", "shell", "i_mag", "j_mag", "i_poscar", "j_poscar", "shift_a", "shift_b", "shift_c", "distance_A"]
    bond_rows = []
    for b in bonds:
        bond_rows.append([
            b.pair_label, b.shell, b.i, b.j, mag_atoms[b.i].index0 + 1, mag_atoms[b.j].index0 + 1,
            b.shift[0], b.shift[1], b.shift[2], f"{b.distance:.8f}"
        ])
    write_csv(os.path.join(args.outdir, "bonds_used_for_J.csv"), bond_header, bond_rows)

    # Shell summary
    summary = []
    for p in sorted(set(b.pair_label for b in bonds)):
        for s in range(1, args.max_shells + 1):
            bl = [b for b in bonds if b.pair_label == p and b.shell == s]
            if not bl:
                continue
            ds = [b.distance for b in bl]
            summary.append([p, s, len(bl), f"{min(ds):.8f}", f"{max(ds):.8f}", f"{np.mean(ds):.8f}"])
    write_csv(os.path.join(args.outdir, "neighbor_shell_summary.csv"),
              ["pair_label", "shell", "n_unique_bonds", "dist_min_A", "dist_max_A", "dist_mean_A"], summary)

    # Config coefficients
    header = ["config"] + atom_labels + params
    rows = []
    for name, spins, coeff in configs:
        rows.append([name] + ["up" if s == 1 else "down" for s in spins] + coeff)
    coeff_path = os.path.join(args.outdir, "J_fit_configurations_coefficients.csv")
    write_csv(coeff_path, header, rows)

    # Equations
    A = np.array([[1] + coeff for _, _, coeff in configs], dtype=float)
    rank = int(np.linalg.matrix_rank(A, tol=1e-10))
    with open(os.path.join(args.outdir, "J_fit_equations.txt"), "w", encoding="utf-8") as f:
        f.write("Hamiltonian convention:\n")
        f.write("E = E0 - sum J_ij S_i S_j, with S_i = +1/-1.\n\n")
        f.write("Unknown vector:\n")
        f.write("[E0, " + ", ".join(params) + "]\n\n")
        f.write(f"Matrix shape: {A.shape[0]} x {A.shape[1]}\n")
        f.write(f"Rank: {rank}\n")
        f.write(f"Full column rank: {rank == A.shape[1]}\n\n")
        for name, _, coeff in configs:
            f.write(equation_string(name, coeff, params) + "\n")

    # Metadata JSON
    meta = {
        "hamiltonian": "E = E0 - sum J_ij S_i S_j",
        "poscar": os.path.abspath(args.poscar),
        "pbc": args.pbc,
        "max_image": args.max_image,
        "max_shells": args.max_shells,
        "shell_tol_A": args.shell_tol,
        "group_by": args.group_by,
        "n_layers": args.n_layers,
        "parameters": params,
        "matrix_shape": list(A.shape),
        "rank": rank,
        "full_column_rank": rank == A.shape[1],
    }
    with open(os.path.join(args.outdir, "metadata.json"), "w", encoding="utf-8") as f:
        json.dump(meta, f, indent=2, ensure_ascii=False)

    print("Done.")
    print(f"Magnetic atoms: {len(mag_atoms)}")
    print(f"Parameters: {params}")
    print(f"Coefficient matrix shape: {A.shape}, rank={rank}, full_column_rank={rank == A.shape[1]}")
    print(f"Output directory: {args.outdir}")
    print(f"Main coefficient file: {coeff_path}")


if __name__ == "__main__":
    main()


```

脚本逻辑：

```shell
1. E1 必定优先生成 FM，全 ↑；
2. 后续优先生成严格 AFM 构型；
   对偶数个磁性原子，要求严格 N/2 ↑、N/2 ↓；
   例如 6 个 Tc 必须是 3↑3↓；
3. 对双层/多组体系，优先生成 layer-AFM，例如 Cr 全 ↑、Ru 全 ↓；
4. 再生成每层内部严格 AFM，例如 Cr 层 3↑3↓、Ru 层全 ↑；
5. 再生成所有层同时严格 AFM，例如 Cr 层 3↑3↓、Ru 层 3↑3↓；
6. 如果这些 AFM 构型仍然无法让矩阵满秩，最后才使用非 AFM 随机构型补足。
```

使用方法，只需要扩胞后的poscar即可：
```shell
python3 1.py POSCAR --mag-elements Cr Ru --group-by element --max-shells 3 --pbc 1 1 0 --n-configs 12
python3 1.py POSCAR --mag-elements Cr Ru --group-by element --max-shells 3 --pbc 1 1 0 --n-configs 12 --shell-tol 0.2
```
--pbc 不用修改，单层体系，块体再改成1 1 1 
--n-configs 是生成需要的磁构型或者是能量方程，原则上几个未知量几个方程，但是最小二乘法拟合应该是更为准确。
